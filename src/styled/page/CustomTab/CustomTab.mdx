### Simple Example
```jsx
import {CustomTab} from 'switchless';
export default function MainComponent() {
  const tabs = [
    {name: 'Overview', slug: 'overview', body: <div>Overview Tab</div>},
    {name: 'Insights', slug: 'insights', body: <div>Insights Tab</div>, stats: '100'},
    {name: 'Analytics', slug: 'analytics', body: <div>Analytics Tab</div>},
  ]

    return <CustomTab
        tabs={tabs}
        indicatorPlacement="top"
    />
}
```

### Nested Tab Example
```jsx
'use client'
import { CustomTab } from 'switchless'

export default function TestPage(){

    const nestedTabs = [
        {name:'Process details', slug:'process', stats:1, body:<>This is a body</>},
        {name:'Process stats', slug:'process_stats', stats:2, body:<>This is another body body</>},
    ]
	const tabs = [
		{
            name:'Overall', 
            slug:'overall', 
            stats:1, 
            body:<>This is a body</>
        },
		{
            name:'Process', 
            slug:'process', 
            stats:2, 
            body:<><CustomTab tabs={nestedTabs} activeTab='process_stats' tabName='process'/></>
        },
	]
	return(<>
		<CustomTab variant='solid' tabs={tabs} activeTab='overall'/>
	</>)
}
```
